Module Odoc_xref2.Lookup_failures

Report non-fatal errors

type kind = [
| `Root
| `Internal
| `Warning
];

`Root failures won't be turned into fatal warnings. `Internal is for lookup failures other than root modules and `Warning for messages to the users. They may be turned into fatal warnings depending on ~warn_error.

type with_failures('a) ;

A value that may be partially unresolved due to failures.

let catch_failures: (unit => 'a) => with_failures ('a);
let report: ?kind: kind => Stdlib.format4 ('fmt,  Stdlib.Format.formatter, unit , unit) => 'fmt;

Report a lookup failure to the enclosing catch_failures call.

let report_important: ?kind: kind => exn => Stdlib.format4 ('fmt,  Stdlib.Format.formatter, unit , unit) => 'fmt;

Like report above but may raise the exception exn if strict mode is enabled

let with_location: Odoc_model.Location_.span => (unit => 'a) => 'a;

Failures reported indirectly by this function will have a location attached.

let handle_failures: warn_error:bool => filename:string => with_failures ('a) => Result.result ('a,  [> `Msg(string) ]) ;

Print failures to stderr. Some failures may be turned into warnings, Error.handle_warnings will be called to print them.
