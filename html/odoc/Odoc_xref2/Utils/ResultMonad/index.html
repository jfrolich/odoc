Module Utils.ResultMonad

The result type and a bind operator. This module is meant to be opened.

type result('a, 'b) = Result.result ('a,  'b) =
| Ok( 'a)
| Error( 'b)
;

Re-export for compat

let map_error: ( 'a => 'b) => result ('c,  'a) => result ('c,  'b) ;
let of_option: error:'a => option('b) => result ('b,  'a) ;
let bind: result ('a,  'b) => ( 'a => result ('c,  'b) ) => result ('c,  'b) ;
let (>>=): result ('a,  'b) => ( 'a => result ('c,  'b) ) => result ('c,  'b) ;
